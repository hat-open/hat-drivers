295    async def _receive_loop(self):
296        try:
297            data = bytearray()
298            while True:
299                spdu_bytes = await self._conn.receive()
300                spdu = encoder.decode(memoryview(spdu_bytes))
301
302                if spdu.type == common.SpduType.DT:
303                    data.extend(spdu.data)
304
305                    if spdu.end is None or spdu.end:
306                        await self._receive_queue.put(data)
307                        data = bytearray()
308
309                elif spdu.type == common.SpduType.FN:
310                    self._close(_dn_spdu)
311                    break
312
313                elif spdu.type == common.SpduType.AB:
314                    self._close(None)
315                    break
316
317                else:
318                    self._close(_ab_spdu)
319                    break
320
321        except ConnectionError:
322            pass
323
324        except Exception as e:
325            self._log.error("receive loop error: %s", e, exc_info=e)
326
327        finally:
328            self.close()

331    async def _send_loop(self):
332        future = None
333        try:
334            while True:
335                data, future = await self._send_queue.get()
336
337                if data is None:
