238    def __init__(self,
239                 conn: acse.Connection,
240                 request_cb: RequestCb,
241                 unconfirmed_cb: UnconfirmedCb):
242        self._conn = conn
243        self._request_cb = request_cb
244        self._unconfirmed_cb = unconfirmed_cb
245        self._loop = asyncio.get_running_loop()
246        self._next_invoke_ids = itertools.count(0)
247        self._response_futures = {}
248        self._close_pdu = 'conclude-RequestPDU', None
249        self._async_group = aio.Group()
250        self._log = _create_connection_logger_adapter(conn.info)

253        self.async_group.spawn(self._receive_loop)
254        self.async_group.spawn(aio.call_on_done, conn.wait_closing(),

262    @property

267    async def send_unconfirmed(self, unconfirmed: common.Unconfirmed):
268        """Send unconfirmed message"""
269        if not self.is_open:
270            raise ConnectionError()
271
272        pdu = 'unconfirmed-PDU', {
273            'service': encoder.encode_unconfirmed(unconfirmed)}
274        data = _mms_syntax_name, _encode(pdu)
275        await self._conn.send(data)
276
277    async def send_confirmed(self,
278                             req: common.Request
279                             ) -> common.Response | common.Error:
280        """Send confirmed request and wait for response"""
281        if not self.is_open:
282            raise ConnectionError()
283
284        invoke_id = next(self._next_invoke_ids)
285        pdu = 'confirmed-RequestPDU', {
286            'invokeID': invoke_id,
287            'service': encoder.encode_request(req)}
288        data = _mms_syntax_name, _encode(pdu)
289        await self._conn.send(data)
290
291        if not self.is_open:
292            raise ConnectionError()
293
294        try:
295            future = self._loop.create_future()
296            self._response_futures[invoke_id] = future
297            return await future
298
299        finally:
300            self._response_futures.pop(invoke_id, None)
301
302    async def _on_close(self):
303        if self._conn.is_open:
304            try:
